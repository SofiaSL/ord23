```bash
$ sudo apt install build-essential git cmake libgmp-dev

$ cd ord23

$ mkdir build

$ cd build

$ cmake -DCMAKE_BUILD_TYPE=Release -DCMAKE_POSITION_INDEPENDENT_CODE=ON ..

$ make -j9

$ ./tests

$ ./ord23
```
